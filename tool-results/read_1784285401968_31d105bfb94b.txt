   149→  const fetchDashboardStats = async () => {
   150→    try {

   152→      const { data: leads, error: leadsError } = await supabase
   153→        .from('leads')
   154→        .select('id, workflow_stage, created_at, monthly_bill');
   155→

   160→        .from('proposals')

   163→      
   164→      if (proposalsError) throw proposalsError;
   165→
   166→      // Calculate stats
   167→      const totalLeads = leads?.length || 0;
   168→      const closedWonLeads = leads?.filter(l => l.workflow_stage === 'completed').length || 0;
   169→      const conversionRate = totalLeads > 0 ? ((closedWonLeads / totalLeads) * 100).toFixed(0) : '0';
   170→      
   171→      // Calculate avg deal size from accepted proposals
   172→      const acceptedProposals = proposals || [];
   173→      const totalDealValue = acceptedProposals.reduce((sum, p) => sum + (p.net_cost || 0), 0);
   174→      const avgDealSize = acceptedProposals.length > 0 
   175→        ? Math.round(totalDealValue / acceptedProposals.length) 
   176→        : 0;
   177→      
   178→      const pendingLeads = leads?.filter(l => 
   179→        l.workflow_stage === 'new' || l.workflow_stage === 'survey'
   180→      ).length || 0;
   181→
   182→      // Calculate trends (last 7 days vs previous 7 days)
   183→      const now = new Date();
   184→      const sevenDaysAgo = new Date(now.getTime() - 7 * 24 * 60 * 60 * 1000);
   185→      const fourteenDaysAgo = new Date(now.getTime() - 14 * 24 * 60 * 60 * 1000);
   186→
   187→      const recentLeads = leads?.filter(l => new Date(l.created_at) >= sevenDaysAgo).length || 0;
   188→      const previousLeads = leads?.filter(l => 
   189→        new Date(l.created_at) >= fourteenDaysAgo && new Date(l.created_at) < sevenDaysAgo
   190→      ).length || 0;

   192→      const leadsTrendValue = previousLeads > 0

   208→      });
